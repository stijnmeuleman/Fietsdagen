<body>

    <div class="container">

        <h1>Activiteit: Fietsen naar werk</h1>

        <div>

            <button class="nav-button" id="prevYear">Vorig Jaar</button>

            <button class="nav-button" id="prevMonth">Vorige Maand</button>

            <span id="currentDate"></span>

            <button class="nav-button" id="nextMonth">Volgende Maand</button>

            <button class="nav-button" id="nextYear">Volgend Jaar</button>

        </div>

        <div id="buttons-container"></div>

    </div>


    <script>

        let currentYear = new Date().getFullYear();

        let currentMonth = new Date().getMonth();

        

        const buttonsContainer = document.getElementById('buttons-container');

        const currentDateSpan = document.getElementById('currentDate');


        function updateCalendar() {

            buttonsContainer.innerHTML = '';

            currentDateSpan.innerText = `${currentYear}-${String(currentMonth + 1).padStart(2, '0')}`;


            const daysInMonth = new Date(currentYear, currentMonth + 1, 0).getDate();


            for (let day = 1; day <= daysInMonth; day++) {

                const date = new Date(currentYear, currentMonth, day);

                const dayOfWeek = date.toLocaleDateString('nl-NL', { weekday: 'long' });

                

                const button = document.createElement('button');

                button.innerText = `${dayOfWeek} ${day}`;

                button.classList.add('day-button');

                button.addEventListener('click', () => {

                    button.classList.toggle('done');

                    button.innerText = `${dayOfWeek} ${day}: ${button.classList.contains('done') ? 'Gedaan' : 'Niet gedaan'}`;

                });

                buttonsContainer.appendChild(button);

            }

        }


        document.getElementById('prevYear').addEventListener('click', () => {

            currentYear--;

            updateCalendar();

        });


        document.getElementById('prevMonth').addEventListener('click', () => {

            if (currentMonth === 0) {

                currentMonth = 11;

                currentYear--;

            } else {

                currentMonth--;

            }

            updateCalendar();

        });


        document.getElementById('nextMonth').addEventListener('click', () => {

            if (currentMonth === 11) {

                currentMonth = 0;

                currentYear++;

            } else {

                currentMonth++;

            }

            updateCalendar();

        });


        document.getElementById('nextYear').addEventListener('click', () => {

            currentYear++;

            updateCalendar();

        });


        updateCalendar();

    </script>
